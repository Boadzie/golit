import golit as gt

# Define a reactive node
@gt.node
def user_metric(user_id: int):
    data = db.fetch_user(user_id)
    return data.metrics

# Dependent node (reactive DAG link)
@gt.node
def dashboard_summary():
    metrics = user_metric(session.current_user)
    return f"User score: {metrics.score}"

# Update and let Golit handle the rest
user_metric.update(user_id=101)
